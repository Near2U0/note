# 函数定义

```c
char s[] = "hello";
int len = strlen(s);
printf("%d\n", len);

//自定义函数
int mystrlen(char s[])//s是形式参数
{//形参在未出现函数调用时，他们并不占用内存，只有在发生函数调用时，形参才被分配内存，函数调用完成之后，形参所占的内存被释放
    char s[] = "hello";
    int len = strlen(s);//s是实参
    //printf("%d\n", len);
    return len;//返回值一定要和函数的定义统一
}
//如果一个函数没有返回值，那么返回类型为void
void test();
//如果一个函数没有明确指定函数的返回类型，那么函数的返回类型是int
test2();//函数声明
test3()//函数定义
{
    return 10;
}


//求两个数的最大者
int max(int a, int b)
{
    return a>b ? a : b;
}

//在C语言中，函数要先定义，然后才能使用，但是有些编译器会编译通过;两种方式解决：在函数使用之前就定义这个函数，或者在函数使用之前先声明这个函数，如下
int max(int a, int b);//函数的声明
int mystrlen(char s[]);
int main()
{
    max(222, 11);
}

//值传递与引用传递
void mystrcat(char s1[], char s2[])
{
    int len1 = 0;
    while(s1[len1++]);
    len1--;
    
    int len2 = 0;
    while(s2[len1++]);
    len2--;
    
    int i;
    for(i=0; i<len2; i++)
    {
        s1[len1+i] = s2[i];
    }
}

//size_t : unsigned int ,如果确认不会返回小于0的值，那么使用此
char s[100] = "hello world";
size_t s = sizeof(s);
size_t len = strlen(s);
```
