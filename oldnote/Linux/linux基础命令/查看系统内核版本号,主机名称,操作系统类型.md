tags: [Linux基础命令]
---


```
#版本
[root@lamp01 ~]# cat /etc/redhat-release
CentOS release 6.3 (Final)

#内核
[root@lamp01 ~]# uname -r
2.6.32-279.el6.i686

#32位还是64位
[root@lamp01 ~]# uname -m
i686

#all
[root@lamp01 ~]# uname -a
Linux lamp01 2.6.32-279.el6.i686 #1 SMP Fri Jun 22 10:59:55 UTC 2012 i686 i686 i386 GNU/Linux

#主机名
[root@lamp01 ~]# hostname
lamp01

```
