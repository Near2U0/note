```
#打印指定用户
[root@lamp01 chenyansong]# id chenyansong
uid=500(chenyansong) gid=500(chenyansong) 组=500(chenyansong),502(student_team)

#打印当前用户
[root@lamp01 chenyansong]# id
uid=0(root) gid=0(root) 组=0(root)


```
