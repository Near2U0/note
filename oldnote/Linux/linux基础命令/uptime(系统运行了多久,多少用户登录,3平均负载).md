```
[root@lamp01 chenyansong]uptime
 16:54:10 up 22:56,  2 users,  load average: 0.00, 0.00, 0.00

 16:54:10    #是当前时间
22:56   #是系统运行的时间(22h 56min)
 2 users    #当前登录用户数
load average: 0.00, 0.00, 0.00    #在过去的1 , 5, 15 min 内的平均负载
```
